#!/bin/bash

# ═══════════════════════════════════════════════════════════════════
# ZIP Password Cracker — Parallel Multi-Method
# Authorized pentest — ZIP: target.zip | Wordlist: World list.txt
# ═══════════════════════════════════════════════════════════════════

# Colors
RED="\e[31m"; GREEN="\e[32m"; YELLOW="\e[33m"; BLUE="\e[34m"; CYAN="\e[36m"; BOLD="\e[1m"; RESET="\e[0m"

ZIP_FILE="target.zip"
WORDLIST="World list.txt"

# Validate files
[[ ! -f "$ZIP_FILE" ]] && echo -e "${RED}Error: $ZIP_FILE not found${RESET}" && exit 1
[[ ! -f "$WORDLIST" ]] && echo -e "${RED}Error: $WORDLIST not found${RESET}" && exit 1

echo -e "${CYAN}${BOLD}"
echo "╔══════════════════════════════════════════════════════════════╗"
echo "║        ZIP PASSWORD CRACKER — PARALLEL ATTACK               ║"
echo "║  Target: $ZIP_FILE                  ║"
echo "║  Wordlist: $WORDLIST                  ║"
echo "╚══════════════════════════════════════════════════════════════╝"
echo -e "${RESET}"

WORD_COUNT=$(wc -l < "$WORDLIST" 2>/dev/null || echo "?")
echo -e "${YELLOW}[*] Wordlist size:${RESET} $WORD_COUNT passwords\n"

# ─── 1. SHELL LOOP (unzip) ──────────────────────────────────────
shell_attack() {
    echo -e "${CYAN}[1] Shell loop attack started...${RESET}"
    local start=$(date +%s) count=0
    
    while IFS= read -r pass || [[ -n "$pass" ]]; do
        ((count++))
        [[ $((count % 500)) -eq 0 ]] && printf "\r${YELLOW}[Shell]${RESET} Tried: %d" "$count"
        
        if unzip -P "$pass" -tq "$ZIP_FILE" 2>/dev/null; then
            echo -e "\n\n${GREEN}╔══════════════════════════════════════╗${RESET}"
            echo -e "${GREEN}║  PASSWORD FOUND via shell loop!        ║${RESET}"
            echo -e "${GREEN}║  Password: ${RESET}$pass"
            echo -e "${GREEN}╚══════════════════════════════════════╝${RESET}"
            echo "$pass" > /tmp/zip_password.txt
            kill $$ 2>/dev/null  # Stop all parallel processes
            exit 0
        fi
    done < "$WORDLIST"
    
    local elapsed=$(( $(date +%s) - start ))
    echo -e "\r${RED}[Shell]${RESET} Not found — $count tries in ${elapsed}s"
}

# ─── 2. 7z ATTACK ──────────────────────────────────────────────
seven_attack() {
    if ! command -v 7z &>/dev/null; then
        echo -e "${RED}[7z]    Not installed, skipping${RESET}"
        return
    fi
    
    echo -e "${CYAN}[2] 7z attack started...${RESET}"
    local start=$(date +%s) count=0
    
    while IFS= read -r pass || [[ -n "$pass" ]]; do
        ((count++))
        [[ $((count % 500)) -eq 0 ]] && printf "\r${YELLOW}[7z]${RESET}     Tried: %d" "$count"
        
        if 7z t -p"$pass" "$ZIP_FILE" 2>/dev/null | grep -q "Everything is Ok"; then
            echo -e "\n\n${GREEN}╔══════════════════════════════════════╗${RESET}"
            echo -e "${GREEN}║  PASSWORD FOUND via 7z!                ║${RESET}"
            echo -e "${GREEN}║  Password: ${RESET}$pass"
            echo -e "${GREEN}╚══════════════════════════════════════╝${RESET}"
            echo "$pass" > /tmp/zip_password.txt
            kill $$ 2>/dev/null
            exit 0
        fi
    done < "$WORDLIST"
    
    local elapsed=$(( $(date +%s) - start ))
    echo -e "\r${RED}[7z]${RESET}     Not found — $count tries in ${elapsed}s"
}

# ─── 3. FCRACKZIP ──────────────────────────────────────────────
fcrack_attack() {
    if ! command -v fcrackzip &>/dev/null; then
        echo -e "${RED}[fcrackzip] Not installed, skipping${RESET}"
        return
    fi
    
    echo -e "${CYAN}[3] fcrackzip attack started...${RESET}"
    local start=$(date +%s)
    
    fcrackzip -u -D -p "$WORDLIST" "$ZIP_FILE" 2>&1 | while read -r line; do
        echo -e "\n${GREEN}╔══════════════════════════════════════╗${RESET}"
        echo -e "${GREEN}║  PASSWORD FOUND via fcrackzip!        ║${RESET}"
        echo -e "${GREEN}║  $line${RESET}"
        echo -e "${GREEN}╚══════════════════════════════════════╝${RESET}"
        echo "$line" > /tmp/zip_password.txt
        kill $$ 2>/dev/null
        exit 0
    done
    
    local elapsed=$(( $(date +%s) - start ))
    echo -e "${RED}[fcrackzip] Not found in ${elapsed}s${RESET}"
}

# ─── 4. JOHN THE RIPPER ────────────────────────────────────────
john_attack() {
    if ! command -v zip2john &>/dev/null || ! command -v john &>/dev/null; then
        echo -e "${RED}[John]   Not installed, skipping${RESET}"
        return
    fi
    
    echo -e "${CYAN}[4] John the Ripper attack started...${RESET}"
    local hashfile="/tmp/zip_john_hash_$$.txt"
    local potfile="/tmp/zip_john_pot_$$.txt"
    
    zip2john "$ZIP_FILE" > "$hashfile" 2>/dev/null
    
    local start=$(date +%s)
    result=$(john --wordlist="$WORDLIST" --pot="$potfile" "$hashfile" 2>&1)
    local elapsed=$(( $(date +%s) - start ))
    
    # Check for cracked password
    cracked=$(john --pot="$potfile" --show "$hashfile" 2>/dev/null | grep -v "^$" | grep -v "^:" | head -1)
    
    if [[ -n "$cracked" ]]; then
        echo -e "\n${GREEN}╔══════════════════════════════════════╗${RESET}"
        echo -e "${GREEN}║  PASSWORD FOUND via John!             ║${RESET}"
        echo -e "${GREEN}║  $cracked${RESET}"
        echo -e "${GREEN}╚══════════════════════════════════════╝${RESET}"
        echo "$cracked" > /tmp/zip_password.txt
    else
        echo -e "${RED}[John]   Not found in ${elapsed}s${RESET}"
    fi
    
    rm -f "$hashfile" "$potfile" 2>/dev/null
}

# ─── 5. HASHCAT (GPU) ──────────────────────────────────────────
hashcat_attack() {
    if ! command -v zip2john &>/dev/null || ! command -v hashcat &>/dev/null; then
        echo -e "${RED}[hashcat] Not installed, skipping${RESET}"
        return
    fi
    
    echo -e "${CYAN}[5] hashcat (GPU) attack started...${RESET}"
    local hashfile="/tmp/zip_hashcat_$$.txt"
    
    zip2john "$ZIP_FILE" > "$hashfile" 2>/dev/null
    
    # Auto-detect hash type
    local mode=17200
    grep -q '\$pkzip\$' "$hashfile" && mode=17220
    grep -q '\$zip2\$' "$hashfile" && mode=17210
    
    local start=$(date +%s)
    hashcat -m $mode -a 0 "$hashfile" "$WORDLIST" --potfile-disable --force --status 2>&1 | \
        grep -E "Recovered|Cracked|Session" | head -3
    local elapsed=$(( $(date +%s) - start ))
    
    result=$(hashcat -m $mode -a 0 "$hashfile" "$WORDLIST" --potfile-disable --force --show 2>/dev/null)
    
    if [[ -n "$result" && "$result" != "0" ]]; then
        echo -e "\n${GREEN}╔══════════════════════════════════════╗${RESET}"
        echo -e "${GREEN}║  PASSWORD FOUND via hashcat!          ║${RESET}"
        echo -e "${GREEN}║  $result${RESET}"
        echo -e "${GREEN}╚══════════════════════════════════════╝${RESET}"
        echo "$result" > /tmp/zip_password.txt
    else
        echo -e "${RED}[hashcat] Not found in ${elapsed}s${RESET}"
    fi
    
    rm -f "$hashfile" 2>/dev/null
}

# ─── MAIN: LAUNCH ALL PARALLEL ────────────────────────────────
echo -e "${YELLOW}[*] Launching all 5 attacks in parallel...${RESET}\n"

echo -e "${GREEN}╔══════════════════════════════════════════════╗${RESET}"
echo -e "${GREEN}║  SPEED REFERENCE                            ║${RESET}"
echo -e "${GREEN}║  Shell loop:     ~50 passwords/sec          ║${RESET}"
echo -e "${GREEN}║  7z:             ~30 passwords/sec          ║${RESET}"
echo -e "${GREEN}║  fcrackzip:      ~5,000 passwords/sec       ║${RESET}"
echo -e "${GREEN}║  John:           ~10,000 passwords/sec      ║${RESET}"
echo -e "${GREEN}║  hashcat (GPU):  ~100M+ passwords/sec       ║${RESET}"
echo -e "${GREEN}╚══════════════════════════════════════════════╝${RESET}\n"

# Run all in background
shell_attack &
seven_attack &
fcrack_attack &
john_attack &
hashcat_attack &

# Wait for all to finish
wait

# Final result
echo -e "\n${BLUE}══════════════════════════════════════════════${RESET}"
if [[ -f /tmp/zip_password.txt ]]; then
    echo -e "${GREEN}[+] PASSWORD CRACKED!${RESET}"
    echo -e "${GREEN}[+] Password:${RESET} $(cat /tmp/zip_password.txt)"
else
    echo -e "${RED}[-] Password not found with any method.${RESET}"
    echo -e "${YELLOW}[*] Try a larger wordlist or different approach.${RESET}"
fi
echo -e "${BLUE}══════════════════════════════════════════════${RESET}"
